#!/bin/bash

if [ $# -ne 3 ]; then
    echo "Błąd: należy podać dokładnie trzy argumenty."
    exit 1
fi

# Przypisanie parametrów
param1=$1
param2=$2
param3=$3

# Znalezienie największego
if [ $param1 -ge $param2 ] && [ $param1 -ge $param3 ]; then
    echo "Największy parametr to: $param1"
elif [ $param2 -ge $param1 ] && [ $param2 -ge $param3 ]; then
    echo "Największy parametr to: $param2"
else
    echo "Największy parametr to: $param3"
fi
