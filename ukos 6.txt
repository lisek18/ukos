#!/bin/bash

if [ ! -d /tmp/backup ]; then
    mkdir -p /tmp/backup
fi

user=$(whoami)
data=$(date +%Y-%m-%d)
archive_name="/tmp/backup/${user}_home_${data}.tar.gz"

tar -czf $archive_name /home/$user 2>/dev/null
#jeśli cały katalog to
#tar -czf $backup_file /home 2>/dev/null

echo "Plik zostal utworzony"

fi
